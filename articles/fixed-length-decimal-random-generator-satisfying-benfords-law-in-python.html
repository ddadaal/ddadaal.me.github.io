import random

def possibility_for_n(n):
    if n == 0:
        return 0.0
    else:
        return (n + 1) * (10 ** -n)

def distribution_list(n):
    prob = {}
    for i in range(10):
        total = 0.0
        for j in range(int(10 ** (n-2)), int(10 ** (n-1))):
            num = j * 10 + i
            total += possibility_for_n(j)
        prob[i] = total
    return prob

def search_list(dist):
    cum_sum = []
    current_sum = 0.0
    for k in range(10):
        current_sum += dist[k]
        cum_sum.append(current_sum)
    return cum_sum

def generate_digit(n, search):
    random_num = random.uniform(0, 1)
    for i in range(10):
        if random_num <= search[i]:
            return i
    return 9  # Fallback, though should never reach here

def generate_one(length):
    digits = []
    for _ in range(length):
        digit = generate_digit(_, search_list(distribution_list(_)))
        digits.append(str(digit))
    number = int(''.join(digits))
    return number

def generate_multiple(length, num):
    numbers = []
    for _ in range(num):
        numbers.append(generate_one(length))
    return numbers

import math, random
from functools import reduce

def possibility_for_n(n):
    return 0 if n==0 else math.log(1+1/n,10)

def distribution_list(n):
    result = []
    for i in range(0,10):
        r = 0
        for j in range(int(math.pow(10,n-2)),int(math.pow(10,n-1))):
            r = r + possibility_for_n(j*10+i)
        result.append(r)
return result

def search_list(distlist):
    return list(map(lambda i: sum(distlist[:i+1]), range(0,10)))

def generate_digit(n):
    rand = random.random()
    searchlist = search_list(distribution_list(n))
    if rand <= searchlist[0]:
        return 0
    for index in range(0,10):
        if rand > searchlist[index] and rand <= searchlist[index+1]:
            return index+1

def generate_one(length):
    return reduce(lambda x,y: x*10+y, map(lambda i: generate_digit(i), range(1,length+1)))

def generate_multiple(length, num):
    return [generate_one(length) for i in range(0,num)]
```

import random

def possibility_for_n(n):
    if n == 0:
        return 0.0
    else:
        return (n + 1) * (10 ** -n)

def distribution_list(n):
    prob = {}
    for i in range(10):
        total = 0.0
        for j in range(int(10 ** (n-2)), int(10 ** (n-1))):
            num = j * 10 + i
            total += possibility_for_n(j)
        prob[i] = total
    return prob

def search_list(dist):
    cum_sum = []
    current_sum = 0.0
    for k in range(10):
        current_sum += dist[k]
        cum_sum.append(current_sum)
    return cum_sum

def generate_digit(n, search):
    random_num = random.uniform(0, 1)
    for i in range(10):
        if random_num <= search[i]:
            return i
    return 9  # Fallback, though should never reach here

def generate_one(length):
    digits = []
    for _ in range(length):
        digit = generate_digit(_, search_list(distribution_list(_)))
        digits.append(str(digit))
    number = int(''.join(digits))
    return number

def generate_multiple(length, num):
    numbers = []
    for _ in range(num):
        numbers.append(generate_one(length))
    return numbers
```

def possibility_for_n(n):
    if n == 0:
        return 0.0
    else:
        return (n + 1) * (10 ** -n)

def distribution_list(n):
    prob = {}
    for i in range(10):
        total = 0.0
        for j in range(int(10 ** (n-2)), int(10 ** (n-1))):
            num = j * 10 + i
            total += possibility_for_n(j)
        prob[i] = total
    return prob

def search_list(dist):
    cum_sum = []
    current_sum = 0.0
    for k in range(10):
        current_sum += dist[k]
        cum_sum.append(current_sum)
    return cum_sum

def generate_digit(n, search):
    random_num = random.uniform(0, 1)
    for i in range(10):
        if random_num <= search[i]:
            return i
    return 9  # Fallback, though should never reach here

def generate_one(length):
    digits = []
    for _ in range(length):
        digit = generate_digit(_, search_list(distribution_list(_)))
        digits.append(str(digit))
    number = int(''.join(digits))
    return number

def generate_multiple(length, num):
    numbers = []
    for _ in range(num):
        numbers.append(generate_one(length))
    return numbers
